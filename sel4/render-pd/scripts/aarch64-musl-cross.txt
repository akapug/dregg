# aarch64-musl-cross.txt — meson cross file for building Mesa-lavapipe targeting
# aarch64-unknown-linux-musl (the seL4-musl PD profile). Uses the brew
# aarch64-unknown-linux-musl GCC 15.2.0 cross toolchain (the same one that builds
# the executor-PD's Lean runtime) + the render-PD's own cross-built static LLVM.
#
# The target LLVM is selected via the cross llvm-config wrapper (out/llvm-elf/bin/
# llvm-config), which reports the aarch64-musl static libs — NOT the host brew LLVM.

[binaries]
c       = '/opt/homebrew/opt/aarch64-unknown-linux-musl/bin/aarch64-linux-musl-gcc'
cpp     = '/opt/homebrew/opt/aarch64-unknown-linux-musl/bin/aarch64-linux-musl-g++'
ar      = '/opt/homebrew/opt/aarch64-unknown-linux-musl/bin/aarch64-linux-musl-ar'
strip   = '/opt/homebrew/opt/aarch64-unknown-linux-musl/bin/aarch64-linux-musl-strip'
ld      = '/opt/homebrew/opt/aarch64-unknown-linux-musl/bin/aarch64-linux-musl-ld'
# llvm-config for the TARGET (cross-built static LLVM). Generated by build-llvm-elf.sh
# (the host native llvm-config in /tmp/llvm-native-tblgen reports darwin libs — wrong;
# we point at a wrapper that reports the aarch64-musl libs).
llvm-config = '@LLVM_CONFIG_CROSS@'
# pkg-config: a wrapper restricted to the musl sysroot so Mesa finds no host libs.
pkg-config = '@PKGCONF_WRAP@'

[host_machine]
system     = 'linux'
cpu_family = 'aarch64'
cpu        = 'aarch64'
endian     = 'little'

[properties]
# No target binaries are executed during the Mesa build (codegen runs native).
needs_exe_wrapper = true
# The musl sysroot (brew toolchain bundles its own sysroot under .../toolchain).
sys_root = '/opt/homebrew/opt/aarch64-unknown-linux-musl/toolchain/aarch64-unknown-linux-musl'

[built-in options]
# Static, headless: link the target LLVM statically into the ICD (no DT_NEEDED on
# libLLVM — self-contained for the PD), no exceptions stripping (LLVM needs RTTI/EH).
default_library = 'static'
prefer_static   = true
