mRel_def
⊢ ∀c b s.
    mRel c b s ⇔
    FLOOKUP s.locals «c» = SOME (ValWord (n2w c)) ∧
    FLOOKUP s.locals «b» = SOME (ValWord (n2w b)) ∧ c ≤ 255 ∧ b < 256
mstep_def
⊢ ∀c b. mstep c b = if b < 128 then c else if c < 255 then c + 1 else 255
stepBody_def
⊢ stepBody =
  If (Cmp Less (Var Local «b») (Const 128w))
    (Assign Local «c» (Var Local «c»))
    (If (Cmp Less (Var Local «c») (Const 255w))
       (Assign Local «c» (Op Add [Var Local «c»; Const 1w]))
       (Assign Local «c» (Const 255w)))

signed_lt_n2w64

stepBody_refines_step
⊢ mRel c b s ⇒
  ∃s'.
    evaluate (stepBody,s) = (NONE,s') ∧
    FLOOKUP s'.locals «c» = SOME (ValWord (n2w (mstep c b))) ∧
    mRel (mstep c b) b s'
