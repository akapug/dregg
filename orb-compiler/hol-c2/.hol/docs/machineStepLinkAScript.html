1(* ===========================================================================
2   C2 probe — LINK A for a MACHINE primitive (a stateful transition step).
3
4   C0/C1 dual-emitted and preservation-proved a REGION primitive (a bounds
5   decision over a byte view): `boundScanLinkAScript.sml` proves the emitted
6   bounds `If` refines the Lean model `boundScan`/`c0_encode` against the REAL
7   Pancake source semantics `panSem$evaluate`/`panSem$eval`.
8
9   This file GENERALIZES that Link A from a one-shot bounds decision to a
10   STATEFUL TRANSITION: `State -> Input -> State`. The primitive is one step of
11   a saturating event-counter FSM (model/MachineStep.lean `C2.step`):
12
13       step c b  =  if b < 128 then c                    (* low byte: hold  *)
14                    else if c < 255 then c + 1 else 255   (* high: saturate  *)
15
16   We prove, against real `panSem$evaluate`, that running the emitted `.pnk`
17   step body updates the machine's state local `c` to EXACTLY `n2w (step c b)`,
18   AND re-establishes the state relation `mRel` at the new counter — i.e. the
19   emitted transition refines the Lean transition and the relation is an
20   INVARIANT the step preserves (so the machine can iterate). That invariant
21   carries the saturation MEANING (`mstep_le`: the counter never exceeds 255).
22
23   Scope, honestly: the whole machine `run` folds `step` over an input stream
24   with a `While` loop. Link A for that fold — a loop-invariant induction over
25   `panSem`'s clocked `While` clause (`c_n = FOLDL step c0 inputs`) — is NOT
26   proven here; it is the exact analogue of C1's deferred scan `While` and is
27   named UNCLOSED at the end. What IS closed end-to-end is the SINGLE transition.
28
29   The comparison faithfulness note from C1 carries over verbatim: Pancake `<`
30   is `LessT` -> `Cmp Less`, the SIGNED word comparison. Both operands here
31   (`b < 256`, `c <= 255`) sit in the non-negative signed range, so the same
32   `signed_lt_n2w64` convention lemma discharges both guards.
33   =========================================================================== *)
34
35open HolKernel boolLib bossLib Parse;
36open arithmeticTheory listTheory bitTheory wordsTheory wordsLib;
37open panLangTheory panSemTheory;
38
39val _ = new_theory "machineStepLinkA";
40
41(* ---------------------------------------------------------------------------
42   The Lean SPEC, re-declared in HOL (byte-identical to C2.step over `num`).
43   --------------------------------------------------------------------------- *)
44Definition mstep_def:
45  mstep (c:num) (b:num) =
46    if b < 128 then c else if c < 255 then c + 1 else 255
47End
48
49(* The saturation MEANING (C2.step_le_cap): the transition never lets the
50   counter exceed the cap. This is the load-bearing safety property and it is
51   what keeps `mRel` an invariant across the step. *)
52Theorem mstep_le:
53  !c b. c <= 255 ==> mstep c b <= 255
54Proof
55  rw [mstep_def]
56QED
57
58(* ---------------------------------------------------------------------------
59   The IMPLEMENTATION: the .pnk step body, as a real panLang AST.
60
61     if b < 128 { c = c; }
62     else { if c < 255 { c = c + 1; } else { c = 255; } }
63
64   `Cmp Less` = the SIGNED test Pancake `<` compiles to. Word literals fixed to
65   the word64 x64 target.
66   --------------------------------------------------------------------------- *)
67Definition stepBody_def:
68  stepBody =
69    If (Cmp Less (Var Local (strlit "b")) (Const (128w:word64)))
70       (Assign Local (strlit "c") (Var Local (strlit "c")))
71       (If (Cmp Less (Var Local (strlit "c")) (Const (255w:word64)))
72           (Assign Local (strlit "c")
73                   (Op Add [Var Local (strlit "c"); Const (1w:word64)]))
74           (Assign Local (strlit "c") (Const (255w:word64))))
75End
76
77(* ---------------------------------------------------------------------------
78   The state relation: the local env encodes state `c` and input `b` as words,
79   and the sizes fit the (signed, non-negative) range the guards need.
80   --------------------------------------------------------------------------- *)
81Definition mRel_def:
82  mRel (c:num) (b:num) (s:(64,'ffi) panSem$state) <=>
83    FLOOKUP s.locals (strlit "c") = SOME (ValWord (n2w c)) /\
84    FLOOKUP s.locals (strlit "b") = SOME (ValWord (n2w b)) /\
85    c <= 255 /\ b < 256
86End
87
88(* ---------------------------------------------------------------------------
89   The convention lemma (identical to C1's): on the non-negative signed range
90   the SIGNED word order agrees with the nat order. Discharges both guards.
91   --------------------------------------------------------------------------- *)
92Theorem signed_lt_n2w64:
93  !x y. x < 2n ** 63 /\ y < 2n ** 63 ==>
94        (((n2w x):word64) < n2w y <=> x < y)
95Proof
96  rw [] >>
97  `(2:num) ** 63 < 2 ** 64` by EVAL_TAC >>
98  `x < dimword(:64) /\ y < dimword(:64)` by
99    (`dimword(:64) = 2 ** 64` by EVAL_TAC >> fs [] >>
100     conj_tac >> metis_tac [LESS_TRANS]) >>
101  `~word_msb ((n2w x):word64) /\ ~word_msb ((n2w y):word64)` by
102    (rw [word_msb_n2w] >> irule NOT_BIT_GT_TWOEXP >> fs []) >>
103  rw [WORD_LT, w2n_n2w] >> fs []
104QED
105
106(* ---------------------------------------------------------------------------
107   Guard evaluations: real `panSem$eval` of each guard = 1w EXACTLY on the arm
108   the Lean SPEC takes. `eval_class_guard` decides hold-vs-event; `eval_cap_guard`
109   decides advance-vs-saturate.
110   --------------------------------------------------------------------------- *)
111Theorem eval_class_guard:
112  mRel c b s ==>
113  eval s (Cmp Less (Var Local (strlit "b")) (Const (128w:word64)))
114    = SOME (ValWord (if b < 128 then 1w else 0w))
115Proof
116  strip_tac >>
117  qpat_x_assum `mRel _ _ _`
118    (strip_assume_tac o SIMP_RULE std_ss [mRel_def]) >>
119  `(128w:word64) = n2w 128` by EVAL_TAC >>
120  `(n2w b:word64 < n2w 128) = (b < 128)`
121     by (irule signed_lt_n2w64 >> fs []) >>
122  simp [eval_def, OPT_MMAP_def, wordLangTheory.word_op_def,
123        asmTheory.word_cmp_def] >> fs []
124QED
125
126Theorem eval_cap_guard:
127  mRel c b s ==>
128  eval s (Cmp Less (Var Local (strlit "c")) (Const (255w:word64)))
129    = SOME (ValWord (if c < 255 then 1w else 0w))
130Proof
131  strip_tac >>
132  qpat_x_assum `mRel _ _ _`
133    (strip_assume_tac o SIMP_RULE std_ss [mRel_def]) >>
134  `(255w:word64) = n2w 255` by EVAL_TAC >>
135  `(n2w c:word64 < n2w 255) = (c < 255)`
136     by (irule signed_lt_n2w64 >> fs []) >>
137  simp [eval_def, OPT_MMAP_def, wordLangTheory.word_op_def,
138        asmTheory.word_cmp_def] >> fs []
139QED
140
141(* ---------------------------------------------------------------------------
142   LINK A, the transition core: real `panSem$evaluate` of the step body updates
143   the state local `c` to EXACTLY the Lean model's next state `n2w (mstep c b)`,
144   with no error. A kernel-checked equation between the actual Pancake source
145   semantics and the Lean transition. This is the stateful generalization of
146   C1's `evaluate_boundsChk` (which wrote a constant sentinel; this writes a
147   data-dependent next state that is a function of the old state).
148   --------------------------------------------------------------------------- *)
149Theorem evaluate_stepBody:
150  mRel c b s ==>
151  evaluate (stepBody, s) =
152    (NONE, set_var (strlit "c") (ValWord (n2w (mstep c b))) s)
153Proof
154  strip_tac >>
155  `eval s (Cmp Less (Var Local (strlit "b")) (Const (128w:word64)))
156     = SOME (ValWord (if b < 128 then 1w else 0w))` by metis_tac [eval_class_guard] >>
157  `eval s (Cmp Less (Var Local (strlit "c")) (Const (255w:word64)))
158     = SOME (ValWord (if c < 255 then 1w else 0w))` by metis_tac [eval_cap_guard] >>
159  `FLOOKUP s.locals (strlit "c") = SOME (ValWord (n2w c))` by fs [mRel_def] >>
160  `(1w:word64) <> 0w` by EVAL_TAC >>
161  (* split the transition's three arms (hold / advance / saturate) and reduce
162     each leaf against real `panSem$evaluate` uniformly. *)
163  Cases_on `b < 128` >> Cases_on `c < 255` >>
164  asm_simp_tac (srw_ss())
165    [stepBody_def, evaluate_def, eval_def, OPT_MMAP_def,
166     wordLangTheory.word_op_def, word_add_n2w, asmTheory.word_cmp_def,
167     is_valid_value_def, lookup_kvar_def, shape_of_def, set_kvar_def,
168     mstep_def]
169QED

171(* ---------------------------------------------------------------------------
172   LINK A, the headline refinement + invariance: after one emitted step,
173     (1) the state local `c` holds exactly the Lean next state, and
174     (2) the state relation `mRel` holds again at the new counter —
175   so the emitted transition REFINES the Lean transition and PRESERVES the
176   relation. (2) carries the saturation MEANING (`c' <= 255`), which is why the
177   step composes into a stream fold. This is the transition-system Link A.
178   --------------------------------------------------------------------------- *)
179Theorem stepBody_refines_step:
180  mRel c b s ==>
181  ?s'. evaluate (stepBody, s) = (NONE, s') /\
182       FLOOKUP s'.locals (strlit "c") = SOME (ValWord (n2w (mstep c b))) /\
183       mRel (mstep c b) b s'
184Proof
185  strip_tac >> drule evaluate_stepBody >> rw [] >>
186  qpat_x_assum `mRel _ _ _`
187    (strip_assume_tac o SIMP_RULE std_ss [mRel_def]) >>
188  simp [mRel_def, set_var_def, finite_mapTheory.FLOOKUP_UPDATE] >>
189  metis_tac [mstep_le]
190QED
