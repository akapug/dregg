1(* C0 probe — HOL4 transcription of the Lean SPEC C0.boundScan, plus
2   kernel-checked EVAL of the reference vectors. This is P2's behavioral
3   round-trip (§1, layer 3) applied to THIS primitive: a HOL4 kernel
4   computation of the same function on the same vectors, to sit beside the
5   Lean model output and the compiled-Pancake machine output. Three
6   independent kernels, one digest per vector.
7
8   Bytes and accumulators are modelled as `num`; the mask 16777216 = 2^24 and
9   the multiplier 31 are literal-for-literal the Pancake `& 16777215` / `* 31`
10   and the Lean `&&& 0xFFFFFF` / `* 31`. (num MOD 2^24 = the low 24 bits, and
11   every intermediate is < 2^24, so num/UInt32/word64 all agree here — the
12   convention seam of P2 §4.2 does not bite because there is no division and no
13   negative value on the path.) *)
14
15open HolKernel boolLib bossLib listTheory arithmeticTheory;
16
17val _ = new_theory "boundScan";
18
19Definition step_def:
20  step acc b = (acc * 31 + b) MOD 16777216
21End
22
23Definition scanFrom_def:
24  (scanFrom a off 0 acc = acc) /\
25  (scanFrom a off (SUC n) acc = scanFrom a (off + 1) n (step acc (EL off a)))
26End
27
28Definition boundScan_def:
29  boundScan a off len =
30    if off + len <= LENGTH a then SOME (scanFrom a off len 0) else NONE
31End
32
33(* "GET / HTTP/1.1\r\n" — byte-identical to C0.arena. *)

36End
