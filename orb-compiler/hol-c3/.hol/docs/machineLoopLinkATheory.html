loopBody_def
⊢ loopBody =
  Seq
    (Assign Local «b» (LoadByte (Op Add [Var Local «base»; Var Local «i»])))
    (Seq stepBody (Assign Local «i» (Op Add [Var Local «i»; Const 1w])))
loopInv_def
⊢ ∀input bs c i s.
    loopInv input bs c i s ⇔
    FLOOKUP s.locals «c» = SOME (ValWord (n2w c)) ∧
    FLOOKUP s.locals «i» = SOME (ValWord (n2w i)) ∧
    FLOOKUP s.locals «len» = SOME (ValWord (n2w (LENGTH input))) ∧
    FLOOKUP s.locals «base» = SOME (ValWord bs) ∧
    (∃bb. FLOOKUP s.locals «b» = SOME (ValWord bb)) ∧ memRel input bs s ∧
    c ≤ 255 ∧ i ≤ LENGTH input ∧ LENGTH input < 2 ** 63 ∧
    EVERY (λx. x < 256) input
machineLoop_def
⊢ machineLoop = While (Cmp Less (Var Local «i») (Var Local «len»)) loopBody
memRel_def
⊢ ∀input bs s.
    memRel input bs s ⇔
    ∀j. j < LENGTH input ⇒
        mem_load_byte s.memory s.memaddrs s.be (bs + n2w j) =
        SOME (n2w input❲j❳)

    loopInv input bs (mstep c input❲i❳) (i + 1) s2

machineLoop_fold_bounded
⊢ ∀k input bs c i s.
    loopInv input bs c i s ∧ LENGTH input − i ≤ k ∧
    LENGTH input − i ≤ s.clock ⇒
    ∃s'.
      evaluate (machineLoop,s) = (NONE,s') ∧
      FLOOKUP s'.locals «c» =
      SOME (ValWord (n2w (FOLDL mstep c (DROP i input))))
machineLoop_refines_run
⊢ loopInv input bs 0 0 s ∧ LENGTH input ≤ s.clock ⇒
  ∃s'.
    evaluate (machineLoop,s) = (NONE,s') ∧
    FLOOKUP s'.locals «c» = SOME (ValWord (n2w (FOLDL mstep 0 input)))
machineLoop_unfold
⊢ loopInv input bs c i s ∧ i < LENGTH input ∧ s.clock ≠ 0 ⇒
  ∃s2.
    evaluate (machineLoop,s) = evaluate (machineLoop,s2) ∧
    loopInv input bs (mstep c input❲i❳) (i + 1) s2 ∧ s2.clock = s.clock − 1
w2w_byte
⊢ ∀x. x < 256 ⇒ w2w (n2w x) = n2w x
