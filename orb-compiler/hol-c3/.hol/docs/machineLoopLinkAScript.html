1(* ===========================================================================
2   C3 probe — LINK A for a LOOP: the fold/scan `While` (C1/C2's UNCLOSED item).
3
4   C1 (`boundScanLinkAScript.sml`) and C2 (`machineStepLinkAScript.sml`) each
5   closed Link A for a SINGLE transition against the real Pancake source
6   semantics `panSem$evaluate`, and each named the same residual UNCLOSED:
7
8     the fold/scan `While` loop + its per-iteration byte-memory `LoadByte`
9     relation — a loop-invariant induction over panSem's CLOCKED `While` clause.
10
11   This file DISCHARGES that loop. We build the machine's stream loop as a real
12   `panLang$prog` `While` that (i) `LoadByte`s the next input byte, (ii) runs the
13   C2 single-step body `stepBody` (opened from `machineStepLinkATheory` and used
14   verbatim — this is the composition mechanism the single-step theorem could not
15   demonstrate), and (iii) advances the index. We prove, against real
16   `panSem$evaluate`, that running the emitted `While` refines the Lean fold
17   `FOLDL mstep` over the input byte stream:
18
19       evaluate (machineLoop, s)  ==>  «c» = n2w (FOLDL mstep c0 input)
20
21   The proof is a loop-invariant induction over the clocked `While`. The
22   invariant is `loopInv` — the C2 state relation lifted to the loop:
23     * `«c»` holds the running fold `n2w (FOLDL mstep 0 (TAKE i input))` (here we
24       carry the running accumulator `c` and prove the residual `DROP i input`),
25     * `«i»` is the loop index, `«len»` the length, `«base»` the buffer address,
26     * `memRel input bs s` — the byte-memory relation: `panSem`'s word-addressed
27       byte memory `mem_load_byte s.memory s.memaddrs s.be (bs + n2w j)` yields
28       `n2w (EL j input)`, the j-th model byte, for every in-range j. This is the
29       LoadByte relation named UNCLOSED in C1 §4-A-3 / C2 §6, here THREADED
30       through the loop (the body writes only locals, so the memory relation is
31       an invariant preserved across every iteration).
32
33   The single step `stepBody_refines_step` is the loop BODY; this induction is
34   what wraps it. The clock accounting is the substance the single-step theorems
35   did not have: each `While` iteration consumes exactly one clock (the body has
36   no `Tick`/`While`/`Call`, so it preserves clock), so `LENGTH input - i` clock
37   suffices — established as a hypothesis and threaded through the induction.
38
39   Faithfulness note carries over verbatim from C1/C2: Pancake `<` is the SIGNED
40   `Cmp Less`, so the guard `i < len` needs the non-negative signed-range side
41   condition `LENGTH input < 2^63` (hence `i < 2^63`); discharged by the reused
42   `signed_lt_n2w64`.
43   =========================================================================== *)

56(* A byte loaded from memory (a word8) widened to the 64-bit machine word is the
57   same nat, provided it is a genuine byte (< 256). This is the width side of the
58   LoadByte relation. *)
59Theorem w2w_byte:
60  !x. x < 256 ==> (w2w ((n2w x):word8) : word64) = n2w x
61Proof
62  rw [w2w_def, w2n_n2w] >>
63  `dimword (:8) = 256` by EVAL_TAC >>
64  fs [] >>
65  `x MOD 256 = x` by (irule LESS_MOD >> fs []) >>
66  fs []
67QED

69(* The list-decomposition the induction turns on: at index i < |l| the residual
70   suffix uncons-es into the current element and the next suffix. *)
71Theorem DROP_EL_CONS_local:
72  !n l. n < LENGTH l ==> DROP n l = EL n l :: DROP (SUC n) l
73Proof
74  Induct >> Cases_on `l` >> fs []
75QED

77(* fix_clock is the identity on a result whose clock did not exceed the input
78   clock (always true for our body — evaluate never raises the clock). Isolating
79   this keeps the clocked-While bookkeeping out of the big simps. *)
80Theorem fix_clock_id:
81  !old res ns. ns.clock <= old.clock ==> fix_clock old (res,ns) = (res,ns)
82Proof
83  rw [fix_clock_def] >>
84  `~(old.clock < ns.clock)` by fs [] >>
85  simp [state_component_equality]
86QED

88(* Sequencing two clock-preserving NONE-returning statements: the panSem `Seq`
89   `fix_clock` collapses to the identity, so the two states just chain. This
90   isolates the only subtle bit of the body evaluation (the fix_clock inside
91   `evaluate (Seq ...)`). *)
92Theorem Seq_NONE:
93  !p1 p2 s sa sb.
94    evaluate (p1,s) = (NONE,sa) /\ sa.clock = s.clock /\
95    evaluate (p2,sa) = (NONE,sb) ==>
96    evaluate (Seq p1 p2, s) = (NONE, sb)
97Proof
98  rpt strip_tac >>
99  simp [evaluate_def] >>
100  `fix_clock s (evaluate (p1,s)) = (NONE, sa)`
101     by (`~(s.clock < sa.clock)` by fs [] >>
102         simp [fix_clock_def, state_component_equality]) >>
103  simp []
104QED

106(* ---------------------------------------------------------------------------
107   The IMPLEMENTATION: the .pnk stream loop, as a real panLang AST.
108
109     while (i < len) {
110       b = ldb (base + i);     // LoadByte at the i-th buffer address
111       <stepBody>              // the C2 single-step FSM body (verbatim)
112       i = i + 1;
113     }
114
115   `stepBody` is IMPORTED from machineStepLinkATheory — the exact term whose
116   single-step refinement C2 proved. `Cmp Less` = the SIGNED guard Pancake `<`
117   compiles to.
118   --------------------------------------------------------------------------- *)
119Definition loopBody_def:
120  loopBody =
121    Seq (Assign Local (strlit "b")
122           (LoadByte (Op Add [Var Local (strlit "base"); Var Local (strlit "i")])))
123    (Seq stepBody
124         (Assign Local (strlit "i")
125            (Op Add [Var Local (strlit "i"); Const (1w:word64)])))
126End
127
128Definition machineLoop_def:
129  machineLoop =
130    While (Cmp Less (Var Local (strlit "i")) (Var Local (strlit "len")))
131          loopBody
132End

134(* ---------------------------------------------------------------------------
135   The byte-memory relation (the LoadByte relation, C1 §4-A-3 / C2 §6): real
136   `panSem` word-addressed byte memory, read at the i-th buffer address, yields
137   the i-th model byte. Stated at `mem_load_byte` — the endianness/packing that
138   POPULATES the memory is the FFI's job and stays outside the theorem.
139   --------------------------------------------------------------------------- *)
140Definition memRel_def:
141  memRel (input:num list) (bs:word64) (s:(64,'ffi) panSem$state) <=>
142    !j. j < LENGTH input ==>
143        mem_load_byte s.memory s.memaddrs s.be (bs + n2w j)
144          = SOME ((n2w (EL j input)):word8)
145End
146
147(* ---------------------------------------------------------------------------
148   The loop invariant: the C2 state relation lifted to the loop. Carries the
149   running accumulator `c` at `«c»`, the index `i` at `«i»`, the length and base,
150   a declared byte slot `«b»`, the byte-memory relation, and the signed-range +
151   byte-range side conditions the guards and step need.
152   --------------------------------------------------------------------------- *)
153Definition loopInv_def:
154  loopInv (input:num list) (bs:word64) (c:num) (i:num)
155          (s:(64,'ffi) panSem$state) <=>
156    FLOOKUP s.locals (strlit "c")    = SOME (ValWord (n2w c)) /\
157    FLOOKUP s.locals (strlit "i")    = SOME (ValWord (n2w i)) /\
158    FLOOKUP s.locals (strlit "len")  = SOME (ValWord (n2w (LENGTH input))) /\
159    FLOOKUP s.locals (strlit "base") = SOME (ValWord bs) /\
160    (?bb. FLOOKUP s.locals (strlit "b") = SOME (ValWord bb)) /\
161    memRel input bs s /\
162    c <= 255 /\ i <= LENGTH input /\ LENGTH input < 2n ** 63 /\
163    EVERY (\x. x < 256) input
164End
165
166(* The invariant depends only on locals/memory, not the clock, so lowering the
167   clock (what the `While` does each iteration) preserves it. *)
168Theorem loopInv_clock:
169  loopInv input bs c i s ==> loopInv input bs c i (s with clock := ck)
170Proof
171  rw [loopInv_def, memRel_def]
172QED

174(* ---------------------------------------------------------------------------
175   The loop guard: real `panSem$eval` of `i < len` = 1w EXACTLY when the index is
176   still in range. Reuses `signed_lt_n2w64` (the guard is the SIGNED `Cmp Less`).
177   --------------------------------------------------------------------------- *)
178Theorem eval_loop_guard:
179  loopInv input bs c i s ==>
180    eval s (Cmp Less (Var Local (strlit "i")) (Var Local (strlit "len")))
181      = SOME (ValWord (if i < LENGTH input then 1w else 0w))
182Proof
183  strip_tac >>
184  fs [loopInv_def] >>
185  `i < 2n ** 63` by (irule LESS_EQ_LESS_TRANS >> qexists_tac `LENGTH input` >> fs []) >>
186  `(n2w i:word64 < n2w (LENGTH input)) = (i < LENGTH input)`
187     by (irule signed_lt_n2w64 >> fs []) >>
188  simp [eval_def, asmTheory.word_cmp_def]
189QED
190
191(* ---------------------------------------------------------------------------
192   The per-iteration byte read: real `panSem$eval` of the `LoadByte (base + i)`
193   returns the i-th model byte, via `memRel` + the width lemma `w2w_byte`.
194   --------------------------------------------------------------------------- *)
195Theorem eval_loadbyte:
196  loopInv input bs c i s /\ i < LENGTH input ==>
197    eval s (LoadByte (Op Add [Var Local (strlit "base"); Var Local (strlit "i")]))
198      = SOME (ValWord ((n2w (EL i input)):word64))
199Proof
200  strip_tac >>
201  `EL i input < 256` by (fs [loopInv_def, EVERY_EL]) >>
202  `mem_load_byte s.memory s.memaddrs s.be (bs + n2w i)
203      = SOME ((n2w (EL i input)):word8)`
204     by (fs [loopInv_def, memRel_def]) >>
205  fs [loopInv_def] >>
206  simp [eval_def, OPT_MMAP_def, wordLangTheory.word_op_def, WORD_ADD_0,
207        w2w_byte]
208QED

210(* ---------------------------------------------------------------------------
211   The loop BODY, one iteration: real `panSem$evaluate` of `loopBody` reads the
212   i-th byte into `«b»`, runs the C2 single step (writing `n2w (mstep c byte)`
213   into `«c»` via `evaluate_stepBody`), advances `«i»`, preserves the clock, and
214   RE-ESTABLISHES the invariant at `(mstep c byte, i+1)`. This is where the C2
215   single-step theorem is COMPOSED: `stepBody`'s refinement is used verbatim as
216   the body's middle statement, and the memory relation is threaded (the body
217   writes only locals, so `memRel` survives).
218   --------------------------------------------------------------------------- *)
219Theorem evaluate_loopBody:
220  loopInv input bs c i s /\ i < LENGTH input ==>
221    ?s2. evaluate (loopBody, s) = (NONE, s2) /\ s2.clock = s.clock /\
222         loopInv input bs (mstep c (EL i input)) (i + 1) s2
223Proof
224  strip_tac >>
225  `EL i input < 256` by (fs [loopInv_def, EVERY_EL]) >>
226  drule_all eval_loadbyte >> strip_tac >>
227  (* string-key disequalities used by the FLOOKUP_UPDATE reductions *)
228  `strlit "c" <> strlit "i" /\ strlit "c" <> strlit "b" /\
229   strlit "i" <> strlit "b" /\ strlit "i" <> strlit "c" /\
230   strlit "b" <> strlit "c" /\ strlit "b" <> strlit "i" /\
231   strlit "len" <> strlit "b" /\ strlit "len" <> strlit "c" /\
232   strlit "len" <> strlit "i" /\ strlit "base" <> strlit "b" /\
233   strlit "base" <> strlit "c" /\ strlit "base" <> strlit "i"` by EVAL_TAC >>
234  (* pull the loopInv fields out without dropping the folded relation *)
235  `FLOOKUP s.locals (strlit "c") = SOME (ValWord (n2w c)) /\
236   FLOOKUP s.locals (strlit "i") = SOME (ValWord (n2w i)) /\
237   FLOOKUP s.locals (strlit "len") = SOME (ValWord (n2w (LENGTH input))) /\
238   FLOOKUP s.locals (strlit "base") = SOME (ValWord bs) /\
239   (?bb. FLOOKUP s.locals (strlit "b") = SOME (ValWord bb)) /\
240   c <= 255 /\ memRel input bs s /\ LENGTH input < 2n ** 63 /\
241   EVERY (\x. x < 256) input` by fs [loopInv_def] >>
242  qabbrev_tac `bv = (n2w (EL i input)):word64` >>
243  qabbrev_tac `sA = set_var (strlit "b") (ValWord bv) s` >>
244  qabbrev_tac `sB = set_var (strlit "c")
245                      (ValWord (n2w (mstep c (EL i input)))) sA` >>
246  qabbrev_tac `sC = set_var (strlit "i") (ValWord (n2w (i + 1))) sB` >>
247  (* clocks as SEPARATE, chained assumptions (each ACCEPT-able) *)
248  `sA.clock = s.clock` by simp [Abbr `sA`, set_var_def] >>
249  `sB.clock = sA.clock` by simp [Abbr `sB`, set_var_def] >>
250  `sC.clock = sB.clock` by simp [Abbr `sC`, set_var_def] >>
251  `sC.clock = s.clock` by simp [] >>
252  (* step 1: Assign «b» = the loaded byte *)
253  `evaluate (Assign Local (strlit "b")
254       (LoadByte (Op Add [Var Local (strlit "base"); Var Local (strlit "i")])), s)
255     = (NONE, sA)`
256     by (simp [Once evaluate_def, Abbr `sA`] >>
257         simp [is_valid_value_def, lookup_kvar_def, shape_of_def]) >>
258  (* step 2: the C2 single step, COMPOSED verbatim. Prove the folded equation
259     inside the `by` so the unfolded `drule` fact does NOT leak into context
260     (two equations with the same LHS make `fs`/`simp` loop). *)
261  `mRel c (EL i input) sA`
262     by (simp [mRel_def, Abbr `sA`, set_var_def, FLOOKUP_UPDATE, Abbr `bv`]) >>
263  `evaluate (stepBody, sA) = (NONE, sB)`
264     by (drule evaluate_stepBody >> simp [Abbr `sB`]) >>
265  (* step 3: Assign «i» = i + 1 *)
266  `FLOOKUP sB.locals (strlit "i") = SOME (ValWord (n2w i))`
267     by simp [Abbr `sB`, Abbr `sA`, set_var_def, FLOOKUP_UPDATE] >>
268  `evaluate (Assign Local (strlit "i")
269       (Op Add [Var Local (strlit "i"); Const (1w:word64)]), sB)
270     = (NONE, sC)`
271     by (simp [Once evaluate_def, eval_def, OPT_MMAP_def,
272               wordLangTheory.word_op_def, word_add_n2w, Abbr `sC`] >>
273         simp [is_valid_value_def, lookup_kvar_def, shape_of_def]) >>
274  (* assemble the two Seqs: every hypothesis of Seq_NONE is now a standalone
275     assumption, so discharge by exact ACCEPT (no rewriting, no loop). *)
276  `evaluate (Seq stepBody
277       (Assign Local (strlit "i")
278          (Op Add [Var Local (strlit "i"); Const (1w:word64)])), sA)
279     = (NONE, sC)`
280     by (irule Seq_NONE >> qexists_tac `sB` >>
281         rpt conj_tac >> first_assum ACCEPT_TAC) >>
282  `evaluate (loopBody, s) = (NONE, sC)`
283     by (simp [loopBody_def] >> irule Seq_NONE >> qexists_tac `sA` >>
284         rpt conj_tac >> first_assum ACCEPT_TAC) >>
285  qexists_tac `sC` >>
286  conj_tac >- first_assum ACCEPT_TAC >>
287  conj_tac >- first_assum ACCEPT_TAC >>
288  (* re-establish the invariant at (mstep c (EL i input), i+1) *)
289  `mstep c (EL i input) <= 255` by (irule mstep_le >> fs []) >>
290  `sC.memory = s.memory /\ sC.memaddrs = s.memaddrs /\ sC.be = s.be`
291     by simp [Abbr `sC`, Abbr `sB`, Abbr `sA`, set_var_def] >>
292  `memRel input bs sC` by fs [memRel_def] >>
293  `i + 1 <= LENGTH input` by fs [] >>
294  simp [loopInv_def, Abbr `sC`, Abbr `sB`, Abbr `sA`, set_var_def,
295        FLOOKUP_UPDATE, Abbr `bv`] >>
296  fs []
297QED

299(* One iteration of the emitted `While`: when the guard is live and clock > 0,
300   `evaluate (machineLoop, s)` reduces to `evaluate (machineLoop, s2)` for a state
301   s2 that satisfies the invariant at the NEXT accumulator/index and has spent
302   exactly one clock tick. This packages the clocked-`While` `fix_clock`/dec_clock
303   bookkeeping so the induction proper is clean. *)
304Theorem machineLoop_unfold:
305  loopInv input bs c i s /\ i < LENGTH input /\ s.clock <> 0 ==>
306  ?s2. evaluate (machineLoop, s) = evaluate (machineLoop, s2) /\
307       loopInv input bs (mstep c (EL i input)) (i + 1) s2 /\
308       s2.clock = s.clock - 1
309Proof
310  strip_tac >>
311  (* establish the guard FIRST, while `loopInv .. s` is the only loopInv in
312     scope, so `drule eval_loop_guard` cannot pick a wrong-state instance *)
313  `eval s (Cmp Less (Var Local (strlit "i")) (Var Local (strlit "len")))
314     = SOME (ValWord 1w)` by (drule eval_loop_guard >> fs []) >>
315  `loopInv input bs c i (dec_clock s)`
316     by (simp [dec_clock_def] >> irule loopInv_clock >> fs []) >>
317  drule_all evaluate_loopBody >> strip_tac >>
318  qexists_tac `s2` >>
319  conj_tac
320  >- (
321      (* unfold the LEFT machineLoop only (one While step); keep the right side
322         `evaluate (machineLoop, s2)` folded, then fold the reduced left back.
323         The clean `evaluate_def` While clause has no residual `fix_clock`, so the
324         body result threads directly (s1 = s2 via the loopBody equation). *)
325      CONV_TAC (LAND_CONV
326        (ONCE_REWRITE_CONV [machineLoop_def] THENC
327         ONCE_REWRITE_CONV [evaluate_def])) >>
328      simp [GSYM machineLoop_def]) >>
329  conj_tac >- fs [] >>
330  fs [dec_clock_def]
331QED

333(* ---------------------------------------------------------------------------
334   LINK A FOR THE LOOP — the loop-invariant induction over the clocked `While`.
335
336   With clock >= (number of remaining iterations), running the emitted `While`
337   from an invariant state at (accumulator c, index i) TERMINATES (no TimeOut)
338   and leaves `«c»` holding EXACTLY the Lean fold `n2w (FOLDL mstep c (DROP i
339   input))` — the model's `run` continued from (c,i). Induction on a bound `k`
340   for the remaining count; the clock threads down one per iteration because the
341   body consumes none (packaged in `machineLoop_unfold`).
342   --------------------------------------------------------------------------- *)
343Theorem machineLoop_fold_bounded:
344  !k input bs c i s.
345    loopInv input bs c i s /\ LENGTH input - i <= k /\
346    LENGTH input - i <= s.clock ==>
347    ?s'. evaluate (machineLoop, s) = (NONE, s') /\
348         FLOOKUP s'.locals (strlit "c")
349           = SOME (ValWord (n2w (FOLDL mstep c (DROP i input))))
350Proof
351  Induct_on `k`
352  >- (
353    (* k = 0: no iterations remain, i.e. i >= LENGTH input. Guard false, exit. *)
354    rpt strip_tac >>
355    `~(i < LENGTH input)` by fs [] >>
356    drule eval_loop_guard >> strip_tac >>
357    `DROP i input = []` by (irule DROP_LENGTH_TOO_LONG >> fs []) >>
358    simp [machineLoop_def, Once evaluate_def] >>
359    fs [loopInv_def]) >>
360  (* k -> SUC k *)
361  rpt strip_tac >>
362  Cases_on `i < LENGTH input`
363  >- (
364    (* an iteration runs: step once, then recurse via the IH *)
365    `s.clock <> 0` by fs [] >>
366    drule_all machineLoop_unfold >> strip_tac >>
367    last_x_assum (qspecl_then
368       [`input`,`bs`,`mstep c (EL i input)`,`i + 1`,`s2`] mp_tac) >>
369    impl_tac >- fs [] >>
370    strip_tac >>
371    qexists_tac `s'` >>
372    (* fold: FOLDL mstep c (DROP i input) = FOLDL mstep (mstep c (EL i input))
373       (DROP (i+1) input); normalise SUC i -> i+1 so it meets the IH's index *)
374    `DROP i input = EL i input :: DROP (i + 1) input`
375       by (`DROP i input = EL i input :: DROP (SUC i) input`
376             by (irule DROP_EL_CONS_local >> fs []) >>
377           fs [arithmeticTheory.ADD1]) >>
378    `FOLDL mstep c (DROP i input)
379       = FOLDL mstep (mstep c (EL i input)) (DROP (i + 1) input)`
380       by asm_simp_tac (srw_ss()) [FOLDL] >>
381    fs []) >>
382  (* no iteration: i >= LENGTH input, guard false, exit (same as base) *)
383  drule eval_loop_guard >> strip_tac >>
384  `DROP i input = []` by (irule DROP_LENGTH_TOO_LONG >> fs []) >>
385  simp [machineLoop_def, Once evaluate_def] >>
386  fs [loopInv_def]
387QED

389(* ---------------------------------------------------------------------------
390   The headline, specialized to a whole run from index 0 with accumulator 0:
391   the emitted `While` computes EXACTLY the Lean fold over the entire input
392   stream, given clock >= |input|. `FOLDL mstep 0 input` is the HOL twin of the
393   Lean model's `C2.run input`.
394   --------------------------------------------------------------------------- *)
395Theorem machineLoop_refines_run:
396  loopInv input bs 0 0 s /\ LENGTH input <= s.clock ==>
397  ?s'. evaluate (machineLoop, s) = (NONE, s') /\
398       FLOOKUP s'.locals (strlit "c")
399         = SOME (ValWord (n2w (FOLDL mstep 0 input)))
400Proof
401  strip_tac >>
402  `LENGTH input - 0 <= LENGTH input` by fs [] >>
403  drule machineLoop_fold_bounded >>
404  disch_then (qspec_then `LENGTH input` mp_tac) >>
405  impl_tac >- fs [] >>
406  simp []
407QED
